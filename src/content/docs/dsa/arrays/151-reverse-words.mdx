## Solution

```ts
function reverseWords(s: string): string {
  return s
    .split(" ")
    .filter((word) => word !== "")
    .reverse()
    .join(" ");
}

reverseWords("the sky is blue"); // blue is sky the
reverseWords("  hello world  "); // world hello
reverseWords("a good  example"); // example good a
```
