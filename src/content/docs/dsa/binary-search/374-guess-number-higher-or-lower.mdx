## Solution

```ts
const guess = function (num: number): number {
  return num;
};

function guessNumber(n: number) {
  let left = 1,
    right = n;

  while (left <= right) {
    let middle = Math.floor((left + right) / 2);
    let result = guess(middle);
    if (result === 0) {
      return middle;
    } else if (result === -1) {
      right = middle - 1;
    } else {
      left = middle + 1;
    }
  }
  
}
```
